    id INTEGER PRIMARY KEY AUTOINCREMENT,

    product_qty INTEGER NOT NULL,
    product_desc TEXT NOT NULL,
    product_img MEDIUMBLOB NOT NULL,
    product_price DOUBLE NOT NULL,
    product_type TEXT NOT NULL
);
