CREATE TABLE order (
    cusname TEXT NOT NULL,
    cus_addr TEXT NOT NULL,
    cus_phoneno TEXT NOT NULL,
    orderitem_name TEXT NOT NULL,
    orderitem_qty INTEGER NOT NULL,
    orderitem_totalprice DOUBLE NOT NULL
);
